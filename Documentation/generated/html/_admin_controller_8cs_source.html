170 public async Task<ActionResult> NewUser(NewUserModel viewModel)
171 {
172 if (ModelState.IsValid)
173 {
174 var user = new AppUser();
175 user.IsEnabled = true;
176 user.IsMfaForce = viewModel.IsMfaForce;
177 user.IsLdapLogin = viewModel.IsLdapLogin;
178 user.FirstName = viewModel.FirstName;
179 user.LastName = viewModel.LastName;
180 user.Email = viewModel.Email;
181 user.EmailConfirmed = true;
182 user.Gender = viewModel.Gender;
183 user.Department = viewModel.Department.ToString();
184 user.UserName = viewModel.UserName;
185
186 var result = await _userManager.CreateAsync(user, viewModel.Password);
187 if (result.Succeeded)
188 {
189 // roles
190 foreach (var item in viewModel.Roles)
191 {
192 if (item.IsAssigned)
193 {
194 await _userManager.AddToRoleAsync(user, item.RoleName);
195 }
196 else
197 {
198 await _userManager.RemoveFromRoleAsync(user, item.RoleName);
199 }
200 }
201 }
202
203 ViewData["message"] = "User created success fully.";
204 _logger.LogInformation("AUDIT: " + User.Identity!.Name + " created new user " + user.UserName);
205 }
206 return View(new NewUserModel());
207 }

415 public async Task<IActionResult> EditUser(string id)
416 {
417 var user = await _userManager.FindByIdAsync(id);
418 if (user == null)
419 {
420 return RedirectToAction("Users");
421 }
422 var viewModel = user.Adapt<EditUserViewModel>();
423
424 var userRoles = await _userManager.GetRolesAsync(user);
425 viewModel.Roles = await _roleManager.Roles.Select(s => new AssignRoleViewModel
426 {
427 RoleId = s.Id,
428 RoleName = s.Name,
429 IsAssigned = userRoles.Any(a => a == s.Name)

447 [HttpPost]
448 public async Task<IActionResult> EditUser(EditUserViewModel viewModel)
449 {
450 if (ModelState.IsValid)
451 {
452 var user = await _userManager.FindByIdAsync(viewModel.Id);
453 if (user != null)
454 {
455 if (user.PhoneNumber != viewModel.PhoneNumber && _userManager.Users.Any(a => a.PhoneNumber == viewModel.PhoneNumber))
456 {
457 ModelState.AddModelError(string.Empty, "Phone number already in use.");
458 }
459 else
460 {
461 user.FirstName = viewModel.FirstName;
462 user.LastName = viewModel.LastName;
463 user.UserName = viewModel.UserName;
464 user.Email = viewModel.Email;
465 user.PhoneNumber = viewModel.PhoneNumber;
466 user.Gender = viewModel.Gender;
467 user.Department = viewModel.Department.ToString();
468
469 var result = await _userManager.UpdateAsync(user);
470 if (result.Succeeded)
471 {
472 // Roles
473 foreach (var item in viewModel.Roles)
474 {
475 if (item.IsAssigned)
476 {
477 await _userManager.AddToRoleAsync(user, item.RoleName);
478 }
479 else
480 {
481 await _userManager.RemoveFromRoleAsync(user, item.RoleName);
482 }
483 }
484
485 // Claims
486 var userClaims = await _userManager.GetClaimsAsync(user);
487 var departmentClaim = userClaims.FirstOrDefault(a => a.Type == "Department");
488 var claimsToAdd = new Claim("Department", Enum.GetName(viewModel.Department)!);
489 if (departmentClaim != null)
490 {
491 await _userManager.ReplaceClaimAsync(user, departmentClaim, claimsToAdd);
492 }
493 else
494 {
495 await _userManager.AddClaimAsync(user, claimsToAdd);
496 }
497 _logger.LogInformation("AUDIT: " + User.Identity!.Name + " modified user " + user.UserName);
498 return RedirectToAction("Users");
499 }
500 result.Errors.ToList().ForEach(f => ModelState.AddModelError(string.Empty, f.Description));
501 }
502 }
503 else
504 {
505 ModelState.AddModelError(string.Empty, "User not found.");
506 }
507 }
508
509 return View(viewModel);
510 }
