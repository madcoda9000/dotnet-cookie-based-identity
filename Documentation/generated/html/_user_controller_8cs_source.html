99 {

102 var user = new AppUser
103 {

161 [HttpPost]
162 public async Task<IActionResult> Login(SignInViewModel viewModel)
163 {
164 if (ModelState.IsValid)
165 {
166 var user = await _userManager.FindByNameAsync(viewModel.UserName);
167 if (user != null)
168 {
169 await _signInManager.SignOutAsync();
170
171 var result = await _signInManager.PasswordSignInAsync(user, viewModel.Password, viewModel.RememberMe, true);
172 if(result.Succeeded && user.IsEnabled==false) {
173 await _signInManager.SignOutAsync();
174 _logger.LogWarning("AUDIT: " + user.UserName + " logged in successfully. But account is disabled!");
175 ModelState.AddModelError(string.Empty, _localizer["1"]);
176 }
177 else if(result.Succeeded && user.IsMfaForce==true) {
178 await _userManager.ResetAccessFailedCountAsync(user);
179 await _userManager.SetLockoutEndDateAsync(user, null);
180 _logger.LogInformation("AUDIT: " + user.UserName + " logged in successfully. But MFA is enforced. Redirecting to action: EnforceTwoFactorAuthenticator");
181 return RedirectToAction("EnforceTwoFactorAuthenticator");
182 }
183 else if (result.Succeeded && user.IsMfaForce==false)
184 {
185 await _userManager.ResetAccessFailedCountAsync(user);
186 await _userManager.SetLockoutEndDateAsync(user, null);
187 _logger.LogInformation("AUDIT: " + user.UserName + " logged in successfully.");
188 var returnUrl = TempData["ReturnUrl"];
189 if (returnUrl != null)
190 {
191 return Redirect(returnUrl.ToString() ?? "/");
192 }
193 return RedirectToAction("Index", "Admin");
194 }
195 else if (result.RequiresTwoFactor)
196 {
197 _logger.LogInformation("AUDIT: " + user.UserName + " logged in successfully. But MFA is enabled. Redirecting to action: TwoFactorLogin");
198 return RedirectToAction("TwoFactorLogin", new { ReturnUrl = TempData["ReturnUrl"] });
199 }
200 else if (result.IsLockedOut)
201 {
202 var lockoutEndUtc = await _userManager.GetLockoutEndDateAsync(user);
203 var timeLeft = lockoutEndUtc.Value - DateTime.UtcNow;
204 _logger.LogWarning("AUDIT: " + user.UserName + " logged in successfully. But Account is locked out for at least another " + timeLeft.Minutes + "minutes!");
205 ModelState.AddModelError(string.Empty, _localizer["2", timeLeft.Minutes.ToString()]);
206 }
207 else if (result.IsNotAllowed)
208 {
209 _logger.LogWarning("AUDIT: " + user.UserName + " logged in successfully. But email is not confirmed!");
210 ModelState.AddModelError(string.Empty, _localizer["3"]);
211 }
212 else
213 {
214 _logger.LogWarning("AUDIT: " + user.UserName + " login in failed. Invalid username or password!");
215 ModelState.AddModelError(string.Empty, _localizer["4"]);
216 }
217 }
218 else
219 {
220 _logger.LogWarning("AUDIT: " + "Login in failed. Invalid username or password!");
221 ModelState.AddModelError(string.Empty, _localizer["4"]);
222 }
223 }
224 return View(viewModel);
225 }

277 ModelState.AddModelError(string.Empty, _localizer["5"]);

279 else if (user.TwoFactorType == Models.TwoFactorType.Email)
280 {
281 var result = vieWModel.IsRecoveryCode ? await _signInManager.TwoFactorRecoveryCodeSignInAsync(vieWModel.VerificationCode) : await _signInManager.TwoFactorSignInAsync("Email", vieWModel.VerificationCode, true, false);
282 if (result.Succeeded)
283 {
284 _logger.LogInformation("AUDIT: " + user.UserName + "Email 2fa verified successfully");
285 return Redirect(TempData["ReturnUrl"]?.ToString() ?? "/");

288 ModelState.AddModelError(string.Empty, _localizer["5"]);
