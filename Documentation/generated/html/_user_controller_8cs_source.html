97 public async Task<IActionResult> Register(SignUpViewModel viewModel)
98 {
99 if (ModelState.IsValid)
100 {
101 var user = new AppUser
102 {
103 FirstName = viewModel.FirstName,
104 LastName = viewModel.LastName,
105 UserName = viewModel.UserName,
106 Email = viewModel.Email,
107 Gender = viewModel.Gender,
108 BirthDay = viewModel.BirthDay,

116 if (result.Succeeded)
117 {

120 var confirmationLink = Url.Action("ConfirmEmail", "User", new

122 userId = user.Id,
123 token = confirmationToken
124 }, HttpContext.Request.Scheme);
125
126 await _emailHelper.SendAsync(new()
127 {
128 Subject = "Confirm e-mail",
129 Body = $"Please <a href='{confirmationLink}'>click</a> to confirm your e-mail address.",
130 To = user.Email
131 });

133 return RedirectToAction("Login");
134 }

136 }

159 [HttpPost]
160 public async Task<IActionResult> Login(SignInViewModel viewModel)
161 {
162 if (ModelState.IsValid)
163 {
164 var user = await _userManager.FindByNameAsync(viewModel.UserName);
165 if (user != null)
166 {
167 await _signInManager.SignOutAsync();
168
169 var result = await _signInManager.PasswordSignInAsync(user, viewModel.Password, viewModel.RememberMe, true);
170 if(result.Succeeded && user.IsEnabled==false) {
171 await _signInManager.SignOutAsync();
172 _logger.LogWarning("AUDIT: " + user.UserName + " logged in successfully. But account is disabled!");
173 ModelState.AddModelError(string.Empty, _localizer["1"]);
174 }
175 else if(result.Succeeded && user.IsMfaForce==true) {
176 await _userManager.ResetAccessFailedCountAsync(user);
177 await _userManager.SetLockoutEndDateAsync(user, null);
178 _logger.LogInformation("AUDIT: " + user.UserName + " logged in successfully. But MFA is enforced. Redirecting to action: EnforceTwoFactorAuthenticator");
179 return RedirectToAction("EnforceTwoFactorAuthenticator");
180 }
181 else if (result.Succeeded && user.IsMfaForce==false)
182 {
183 await _userManager.ResetAccessFailedCountAsync(user);
184 await _userManager.SetLockoutEndDateAsync(user, null);
185 _logger.LogInformation("AUDIT: " + user.UserName + " logged in successfully.");
186 var returnUrl = TempData["ReturnUrl"];
187 if (returnUrl != null)
188 {
189 return Redirect(returnUrl.ToString() ?? "/");
190 }
191 return RedirectToAction("Index", "Admin");
192 }
193 else if (result.RequiresTwoFactor)
194 {
195 _logger.LogInformation("AUDIT: " + user.UserName + " logged in successfully. But MFA is enabled. Redirecting to action: TwoFactorLogin");
196 return RedirectToAction("TwoFactorLogin", new { ReturnUrl = TempData["ReturnUrl"] });
197 }
198 else if (result.IsLockedOut)
199 {
200 var lockoutEndUtc = await _userManager.GetLockoutEndDateAsync(user);
201 var timeLeft = lockoutEndUtc.Value - DateTime.UtcNow;
202 _logger.LogWarning("AUDIT: " + user.UserName + " logged in successfully. But Account is locked out for at least another " + timeLeft.Minutes + "minutes!");
203 ModelState.AddModelError(string.Empty, _localizer["2", timeLeft.Minutes.ToString()]);
204 }
205 else if (result.IsNotAllowed)
206 {
207 _logger.LogWarning("AUDIT: " + user.UserName + " logged in successfully. But email is not confirmed!");
208 ModelState.AddModelError(string.Empty, _localizer["3"]);
209 }
210 else
211 {
212 _logger.LogWarning("AUDIT: " + user.UserName + " login in failed. Invalid username or password!");
213 ModelState.AddModelError(string.Empty, _localizer["4"]);
214 }
215 }
216 else
217 {
218 _logger.LogWarning("AUDIT: " + "Login in failed. Invalid username or password!");
219 ModelState.AddModelError(string.Empty, _localizer["4"]);
220 }
221 }
222 return View(viewModel);
223 }

278 {
279 var result = vieWModel.IsRecoveryCode ? await _signInManager.TwoFactorRecoveryCodeSignInAsync(vieWModel.VerificationCode) : await _signInManager.TwoFactorSignInAsync("Email", vieWModel.VerificationCode, true, false);
280 if (result.Succeeded)
